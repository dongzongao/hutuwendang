# 排序算法题

> 排序是算法基础，理解各排序的分治思想、稳定性和复杂度是关键。

---

## 1. 快速排序

**思路**：选 pivot，分区（左边 ≤ pivot，右边 > pivot），递归两侧。平均 O(n log n)，最坏 O(n²)（已排序）。

<script setup>
const quickSortCode = `// 快速排序
function quickSort(nums, left = 0, right = nums.length - 1) {
  if (left >= right) return;
  const pivot = partition(nums, left, right);
  __trace__(\`分区完成: pivot=\${nums[pivot]} 在位置\${pivot}\`, {
    左段: nums.slice(left, pivot),
    pivot: nums[pivot],
    右段: nums.slice(pivot + 1, right + 1)
  });
  quickSort(nums, left, pivot - 1);
  quickSort(nums, pivot + 1, right);
}

function partition(nums, left, right) {
  const pivotVal = nums[right];
  let i = left - 1;
  for (let j = left; j < right; j++) {
    if (nums[j] <= pivotVal) {
      i++;
      [nums[i], nums[j]] = [nums[j], nums[i]];
    }
  }
  [nums[i+1], nums[right]] = [nums[right], nums[i+1]];
  return i + 1;
}

const nums = [3, 6, 8, 10, 1, 2, 1];
console.log('输入:', [...nums]);
quickSort(nums);
console.log('输出:', nums);
`
</script>

## 2. 归并排序

**思路**：分治，不断二分直到单元素，然后两两有序合并。稳定排序，时间复杂度稳定 O(n log n)。

<script setup>
const mergeSortCode = `// 归并排序
function mergeSort(nums, left = 0, right = nums.length - 1) {
  if (left >= right) return;
  const mid = (left + right) >> 1;
  mergeSort(nums, left, mid);
  mergeSort(nums, mid + 1, right);
  merge(nums, left, mid, right);
}

function merge(nums, left, mid, right) {
  const tmp = [];
  let i = left, j = mid + 1;
  while (i <= mid && j <= right) {
    if (nums[i] <= nums[j]) tmp.push(nums[i++]);
    else tmp.push(nums[j++]);
  }
  while (i <= mid) tmp.push(nums[i++]);
  while (j <= right) tmp.push(nums[j++]);
  for (let k = 0; k < tmp.length; k++) nums[left + k] = tmp[k];

  __trace__(\`合并 [\${left}...\${mid}] + [\${mid+1}...\${right}] → [\${tmp.join(',')}]\`, {
    左段: nums.slice(left, mid + 1),
    右段: nums.slice(mid + 1, right + 1),
    合并结果: [...tmp]
  });
}

const nums = [8, 4, 5, 7, 1, 3, 6, 2];
console.log('输入:', [...nums]);
mergeSort(nums);
console.log('输出:', nums);
`
</script>

## 复杂度对比

| 算法 | 平均时间 | 最坏时间 | 空间 | 稳定性 |
|------|----------|----------|------|--------|
| 快速排序 | O(n log n) | O(n²) | O(log n) | 不稳定 |
| 归并排序 | O(n log n) | O(n log n) | O(n) | 稳定 |
| 堆排序 | O(n log n) | O(n log n) | O(1) | 不稳定 |
| 冒泡排序 | O(n²) | O(n²) | O(1) | 稳定 |
| 插入排序 | O(n²) | O(n²) | O(1) | 稳定 |
